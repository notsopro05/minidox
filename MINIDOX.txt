function handleFileUpload(e) {
  const file = e.target.files[0];
  if (!file) return;
  processFile(file);
  e.target.value = '';
}

function processFile(file) {
  state.currentFileName = file.name;
  const ext = file.name.split('.').pop().toLowerCase();
  const sizeKB = Math.round(file.size / 1024);

  showToast('info', `📁 Loading ${file.name}...`);

  if (['png','jpg','jpeg','gif','webp','bmp'].includes(ext)) {
    // Simulate OCR
    setTimeout(() => {
      const mockOCR = `[OCR Extracted from ${file.name}]\n\nThis is simulated OCR content extracted from your image file. In a production environment, this would use a real OCR engine such as Tesseract.js to extract text from the image.\n\nImage file: ${file.name}\nFile size: ${sizeKB} KB\nFormat: ${ext.toUpperCase()}\n\nPlease note that image OCR is simulated in this demo. The actual extracted text would appear here.`;
      loadTextToReader(mockOCR, file.name, sizeKB);
    }, 800);
    return;
  }

  const reader = new FileReader();

  if (ext === 'pdf') {
    reader.onload = () => {
      const simText = `[Extracted from ${file.name}]\n\nThis is simulated PDF text extraction. In production, a PDF parsing library (pdf.js) would extract the actual text content.\n\nFile: ${file.name}\nSize: ${sizeKB} KB\n\nLorem ipsum dolor sit amet, consectetur adipiscing elit. Sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat. Duis aute irure dolor in reprehenderit in voluptate velit esse cillum dolore eu fugiat nulla pariatur. Excepteur sint occaecat cupidatat non proident, sunt in culpa qui officia deserunt mollit anim id est laborum.`;
      loadTextToReader(simText, file.name, sizeKB);
    };
    reader.readAsArrayBuffer(file);
  } else if (['doc','docx'].includes(ext)) {
    reader.onload = () => {
      const simText = `[Extracted from ${file.name}]\n\nThis is simulated Word document extraction. In production, mammoth.js would convert the DOCX to readable text.\n\nFile: ${file.name}\nSize: ${sizeKB} KB\n\nDocument content would appear here with all paragraphs, headings, and formatting preserved as plain text for the reader.`;
      loadTextToReader(simText, file.name, sizeKB);
    };
    reader.readAsArrayBuffer(file);
  } else {
    // Text-based files
    reader.onload = (e) => {
      loadTextToReader(e.target.result, file.name, sizeKB);
    };
    reader.readAsText(file);
  }
}

}

function loadToReader() {
  const text = document.getElementById('inputArea').value.trim();
  if (!text) { showToast('error', 'Please enter or upload some text first.'); return; }
  rehydrateReadingPane(text);
  switchTab('reading', 'reader');
  showToast('success', 'Text loaded to reader!');
}

function rehydrateReadingPane(text) {
  const pane = document.getElementById('readingContent');
  const words = text.split(/(\s+)/);
  state.words = [];
  let html = '';
  let wordIdx = 0;
  words.forEach(part => {
    if (/^\s+$/.test(part)) {
      html += part.replace(/\n/g, '<br>');
    } else if (part.length > 0) {
      html += `<span class="word" data-idx="${wordIdx}" onclick="clickWord(${wordIdx})">${escHtml(part)}</span>`;
      state.words.push(part);
      wordIdx++;
    }
  });
  pane.innerHTML = html || '<em style="color:#9ca3af">Empty document.</em>';
  state.currentWord = -1;
  document.getElementById('wordPos').textContent = '0';
  document.getElementById('wordTotal').textContent = state.words.length;
  applyReadingStyles();
}

function escHtml(t) {
  return t.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

  highlightWordAt(idx);
}

function highlightWordAt(idx) {
  document.querySelectorAll('.word').forEach(w => {
    w.classList.remove('current-highlight');
  });
  const wordEls = document.querySelectorAll('.word');
  if (wordEls[idx]) {
    wordEls[idx].classList.add('current-highlight');
    wordEls[idx].scrollIntoView({ behavior: 'smooth', block: 'center' });
  }
  state.currentWord = idx;
  document.getElementById('wordPos').textContent = idx + 1;
}

function nextWord() {
  if (state.words.length === 0) return;
  const next = (state.currentWord + 1) % state.words.length;
  highlightWordAt(next);
}

function prevWord() {
  if (state.words.length === 0) return;
  const prev = state.currentWord <= 0 ? state.words.length - 1 : state.currentWord - 1;
  highlightWordAt(prev);
}

function resetHighlight() {
  document.querySelectorAll('.word').forEach(w => w.classList.remove('current-highlight'));
  state.currentWord = -1;
  document.getElementById('wordPos').textContent = '0';
  stopAutoHighlight();
}

function toggleAuto() {
  state.autoHighlight = !state.autoHighlight;
  const btn = document.getElementById('autoToggle');
  btn.classList.toggle('on', state.autoHighlight);
  if (state.autoHighlight) startAutoHighlight();
  else stopAutoHighlight();
  showToast('info', state.autoHighlight ? '⚡ Auto-advance ON' : '⏸ Auto-advance OFF');
}

function startAutoHighlight() {
  stopAutoHighlight();
  const speed = parseFloat(document.getElementById('speedInput').value) * 1000;
  state.highlightTimer = setInterval(() => {
    if (state.words.length === 0) return;
    const next = state.currentWord < 0 ? 0 : (state.currentWord + 1) % state.words.length;
    highlightWordAt(next);
    if (next === 0 && state.currentWord > 0) {

function summarizeText() {
  const text = document.getElementById('inputArea').value.trim();
  if (!text) { showToast('error', 'No text to summarize. Load content first.'); return; }
  const n = parseInt(document.getElementById('summaryCount').value) || 4;

  document.getElementById('summaryOutput').innerHTML = summary;
  switchTab('summary', 'reader');
  showToast('success', '🤖 Summary generated!');
}

function extractiveSummarize(text, n) {
  const sentences = text.match(/[^.!?]+[.!?]+/g) || [text];
  if (sentences.length <= n) return sentences.map(s => `<p>${escHtml(s.trim())}</p>`).join('');
  // Score by word frequency
  const wordFreq = {};
  const allWords = text.toLowerCase().split(/\W+/).filter(w => w.length > 3);
  allWords.forEach(w => wordFreq[w] = (wordFreq[w] || 0) + 1);
  const scored = sentences.map((s, i) => {
    const words = s.toLowerCase().split(/\W+/).filter(w => w.length > 3);
    const score = words.reduce((sum, w) => sum + (wordFreq[w] || 0), 0) / Math.max(1, words.length);
    return { s: s.trim(), score, i };
  });
  const top = [...scored].sort((a, b) => b.score - a.score).slice(0, n);
  top.sort((a, b) => a.i - b.i);
  return top.map(t => `<p style="margin-bottom:0.7rem;padding:0.7rem;background:#f9fafb;border-radius:8px;border-left:3px solid #2626e6">${escHtml(t.s)}</p>`).join('');
}

function exportDoc(format) {
  const content = document.getElementById('inputArea').value || document.getElementById('readingContent')?.innerText || '';
  if (!content.trim()) { showToast('error', 'No content to export.'); return; }
  const name = (state.currentFileName || 'minidox_export').replace(/\.[^.]+$/, '');

  if (format === 'copy') {
    navigator.clipboard.writeText(content).then(() => {
      showToast('success', '📋 Copied to clipboard!');
      closeModal('exportModal');
    });
    return;
  }
  let blob, filename;
  if (format === 'txt') {
    blob = new Blob([content], { type: 'text/plain' });
    filename = name + '.txt';
  } else if (format === 'html') {
    const html = `<!DOCTYPE html>\n<html lang="en">\n<head>\n<meta charset="UTF-8">\n<title>${escHtml(name)}</title>\n<style>body{font-family:${FONTS[state.currentFontKey]?.css||'sans-serif'};background:${state.currentTint};color:${state.textColor};max-width:800px;margin:2rem auto;padding:2rem;line-height:${state.lineHeight};letter-spacing:${state.letterSpacing}em;word-spacing:${state.wordSpacing}em;font-size:${state.fontSize}rem}</style>\n</head>\n<body>\n<pre style="white-space:pre-wrap">${escHtml(content)}</pre>\n</body>\n</html>`;
    blob = new Blob([html], { type: 'text/html' });
    filename = name + '.html';
  } else if (format === 'md') {
    blob = new Blob(['# ' + name + '\n\n' + content], { type: 'text/markdown' });
    filename = name + '.md';
  }
  if (blob) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = filename; a.click();
    URL.revokeObjectURL(url);
    showToast('success', `📥 Exported as ${filename}`);
    closeModal('exportModal');
  }
}

function analyzeText() {
  const text = document.getElementById('toolsEditor').value.trim();

  const words = text.split(/\s+/).filter(Boolean);
  const sentences = (text.match(/[^.!?]+[.!?]+/g) || []).length;
  const paragraphs = text.split(/\n\n+/).filter(Boolean).length;
  const avgWordLen = words.reduce((s, w) => s + w.replace(/\W/g, '').length, 0) / Math.max(1, words.length);
  const readingTime = Math.ceil(words.length / 200);
  const freqMap = {};
  words.map(w => w.toLowerCase().replace(/\W/g, '')).filter(w => w.length > 3).forEach(w => freqMap[w] = (freqMap[w] || 0) + 1);
  const topWords = Object.entries(freqMap).sort((a, b) => b[1] - a[1]).slice(0, 5);
  const charCount = text.length;
  const charNoSpace = text.replace(/\s/g, '').length;

  document.getElementById('analysisOutput').innerHTML = `
    <div style="display:grid;grid-template-columns:1fr 1fr;gap:0.7rem">
      ${stat('Words', words.length)}
      ${stat('Characters', charCount)}
      ${stat('Chars (no spaces)', charNoSpace)}
      ${stat('Sentences', sentences)}
      ${stat('Paragraphs', paragraphs)}
      ${stat('Avg word length', avgWordLen.toFixed(1) + ' chars')}
      ${stat('Reading time', readingTime + ' min')}
      ${stat('Unique words', Object.keys(freqMap).length)}
    </div>
    <div style="margin-top:1rem">
      <div style="font-size:0.75rem;font-weight:700;color:#374151;margin-bottom:0.5rem;text-transform:uppercase;letter-spacing:0.05em">Top Words</div>
      <div style="display:flex;flex-wrap:wrap;gap:0.4rem">
        ${topWords.map(([w, c]) => `<span class="badge badge-blue">${escHtml(w)} (${c})</span>`).join('')}
      </div>
    </div>`;
}

  }
}

}

/* Transforms */
function transformText(type) {
  const t = document.getElementById('toolsEditor');
  saveToolsHistory();
  let lines = t.value.split('\n');
  if (type === 'removeSpaces') t.value = t.value.replace(/[ \t]+/g, ' ').replace(/  +/g, ' ');
  else if (type === 'removeDups') { const seen = new Set(); t.value = lines.filter(l => { if (seen.has(l)) return false; seen.add(l); return true; }).join('\n'); }
  else if (type === 'numberLines') t.value = lines.map((l, i) => `${i + 1}. ${l}`).join('\n');
  else if (type === 'reverseLines') t.value = lines.reverse().join('\n');
  else if (type === 'sortLines') t.value = lines.sort((a, b) => a.localeCompare(b)).join('\n');
  else if (type === 'shuffle') { for (let i = lines.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [lines[i], lines[j]] = [lines[j], lines[i]]; } t.value = lines.join('\n'); }
  updateToolsWordCount();
  showToast('success', 'Transform applied!');
}

function mergeSelectedToTools() {
  const checks = document.querySelectorAll('#toolsMergeList input[type=checkbox]:checked');
  if (checks.length < 2) { showToast('error', 'Check at least 2 documents.'); return; }
  const merged = Array.from(checks).map(c => {
    const id = parseInt(c.id.replace('merge-', ''));
    const doc = state.library.find(d => d.id === id);
    return doc ? `=== ${doc.name} ===\n\n${doc.content}` : '';
  }).filter(Boolean).join('\n\n---\n\n');
  document.getElementById('toolsEditor').value = merged;
  updateToolsWordCount();
  showToast('success', `✅ Merged ${checks.length} documents into editor!`);
}

function showToast(type, message) {
  const container = document.getElementById('toastContainer');
  const toast = document.createElement('div');
  toast.className = `toast ${type}`;
